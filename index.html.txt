    <!-- 
        A-Scene setup:
        - embedded: keeps it within the div/body
        - arjs: initializes the AR engine
        - vr-mode-ui: disabled because we are using the phone screen as the view, not a headset
    -->
    <a-scene 
        embedded 
        arjs="sourceType: webcam; debugUIEnabled: false; trackingMethod: best;"
        vr-mode-ui="enabled: false">
        
        <!-- Define the Assets (Materials, Models) -->
        <a-assets>
            <!-- 
                FIX: Added crossorigin="anonymous" to the image tag.
                This is required when loading textures from a different domain
                to satisfy WebGL/Three.js security policies (CORS).
            -->
            <img id="grid" src="https://cdn.aframe.io/a-painter/images/floor.jpg" crossorigin="anonymous">
        </a-assets>

        <!-- 
            A-Marker: The anchor for the AR content.
            "preset: hiro" uses the standard hiro pattern.
        -->
        <a-marker preset="hiro" id="main-marker">
            
            <!-- A 3D Box that appears when the marker is found -->
            <a-box 
                position="0 0.5 0" 
                material="src: #grid; color: #4CC3D9; transparent: true; opacity: 0.9"
                animation="property: rotation; to: 0 360 0; loop: true; dur: 5000; easing: linear">
            </a-box>

            <!-- A floating text label -->
            <a-text 
                value="A-Frame AR" 
                position="0 1.2 0" 
                align="center" 
                color="#FFF" 
                scale="1.5 1.5 1.5"
                geometry="primitive: plane; width: 1.5; height: 0.3" 
                material="color: #333">
            </a-text>

            <!-- A rotating ring around the base -->
            <a-ring 
                color="#EF2D5E" 
                radius-inner="0.6" 
                radius-outer="0.7" 
                rotation="-90 0 0"
                animation="property: scale; from: 1 1 1; to: 1.2 1.2 1.2; dir: alternate; loop: true; dur: 1000">
            </a-ring>

        </a-marker>

        <!-- Static camera that moves according to the marker position -->
        <a-entity camera></a-entity>
        
    </a-scene>

    <script>
        // Simple logic to update the UI status based on marker detection
        const marker = document.querySelector('#main-marker');
        const statusText = document.querySelector('#status');

        marker.addEventListener('markerFound', () => {
            statusText.innerText = "Marker Found!";
            statusText.style.color = "#00ff00";
        });

        marker.addEventListener('markerLost', () => {
            statusText.innerText = "Waiting for Marker...";
            statusText.style.color = "#aaa";
        });
    </script>
